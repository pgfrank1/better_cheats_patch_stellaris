decision_cheatinterface_open_planetmenu = {
	owned_planets_only = no
	icon = decision_consecrated_worlds
	ai_weight = { weight = 0 }
	allow = {
		hidden_trigger = {
			or = {
				is_multiplayer = no
				SOLAR_SYSTEM = {
					OWNER = {
						has_country_flag = hosting
					}
				}
				SOLAR_SYSTEM = {
					OWNER = {
						is_ai = no
						not = { has_country_flag = cheatinterface_is_in_menu }
						or = {
							and = {
								has_country_flag = cheatinterface_allow_planet_colonized
								PREVPREV = {
									OWNER = {
										is_ai = no
									}
								}
							}
							and = {
								has_country_flag = cheatinterface_allow_planet_non_colonized
								PREVPREV = {
									not = {
										OWNER = {
											is_ai = no
										}
									}
								}
							}
						}
					}
				}
			}
		}
		custom_tooltip_success = {
			text = bccheatmenu_no_permission
		}
		custom_tooltip_fail = {
			text = bccheatmenu_no_permission
		}
	}
	effect = {
		custom_tooltip = decision_cheatinterface_open_planetmenu_ct
		hidden_effect = {
			SOLAR_SYSTEM = {
				OWNER = {
					country_event = {
						id = bcplanetmenu.1000
					}		
				}
			}
		}
	}

}

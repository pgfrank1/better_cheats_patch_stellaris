country_event = {
	id = bcplanetmenu.1
	title = "bcplanetmenu_1_title"
	desc = "bcplanetmenu_1_desc"
	is_triggered_only = yes
	diplomatic = yes
	picture_event_data = { 
		portrait = owner.ruler
		planet_background = event_target:planet_selected
		room = cheatinterface_planetmenu_room
	}

	immediate = {
		set_country_flag = cheatinterface_is_in_menu
	}

	#### Planet Class ####
	option = {
		name = "bcplanetmenu_10_title"
		hidden_effect = {
			country_event = {
				id = bcplanetmenu.10
			}
		}
	}
	#### Planet Size ####
	option = {
		name = "bcplanetmenu_20_title"
		hidden_effect = {
			country_event = {
				id = bcplanetmenu.20
			}
		}
	}
	#### Planet Modifier ####
	option = {
		name = "bcplanetmenu_30_title"
		trigger = { always = no }
		hidden_effect = {
			country_event = {
				id = bcplanetmenu.30
			}
		}
	}
	#### Planet Features ####
	option = {
		name = "bcplanetmenu_40_title"
		hidden_effect = {
			country_event = {
				id = bcplanetmenu.40
			}
		}
	}
	#### Enable planet ring ####
	option = {
		name = "bcplanetmenu_1_option_2"
		trigger = {
			event_target:planet_selected = {
				has_ring = no
			}
		}
		hidden_effect = {
			event_target:planet_selected = {
				set_ring = yes
			}
			country_event = {
				id = bcplanetmenu.1
			}
		}
	}
	#### Disable planet ring ####
	option = {
		name = "bcplanetmenu_1_option_2_1"
		trigger = {
			event_target:planet_selected = {
				has_ring = yes
			}
		}
		hidden_effect = {
			event_target:planet_selected = {
				set_ring = no
			}
			country_event = {
				id = bcplanetmenu.1
			}
		}
	}
	#### Colonize Planet ####
	option = {
		name = "bcplanetmenu_1_option_1"
		allow = {
			hidden_trigger = {
				or = {
					has_country_flag = cheatinterface_allow_planet_colonized
					is_multiplayer = no
					has_country_flag = cheatinterface_hosting
				}
				event_target:planet_selected = {
					not = { OWNER = { is_ai = no } }
				}
			}
		}
		hidden_effect = {
			country_event = {
				id = bcplanetmenu.1
			}
			event_target:planet_selected = {
				start_colony = {
					owner = PREV
					species = owner_main_species
				}
				add_colony_progress = 0.999
			}
		}
	}
	# Close # 
	option = {
		name = "bccheatmenu_close"
		default_hide_option = yes
		hidden_effect = {
			remove_country_flag = cheatinterface_is_in_menu
		}
	}
}
